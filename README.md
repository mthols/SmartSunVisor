ESP32-WROOM-32E Pinout
-----------------------------------------
#### Photovoltaic Cells

ADC1_CH6 -> IO34 -> Cell 1

ADC1_CH7 -> IO35 -> Cell 2

ADC1_CH4 -> IO32 -> Cell 3

ADC1_CH5 -> IO33 -> Cell 4

ADC2_CH8 -> IO25 -> Cell 5

ADC2_CH9 -> IO26 -> Cell 6

ADC2_CH7 -> IO27 -> Cell 7

ADC2_CH6 -> IO14 -> Cell 8

#### Track Motor

IO16 -> Direction

IO17 -> Step

#### Visor Motor

IO5 -> Direction

IO18 -> Step

#### Limit Switches

IO12 -> Door window

IO13 -> Windshield
